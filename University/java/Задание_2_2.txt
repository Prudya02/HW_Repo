import java.util.Arrays;
import java.util.Scanner;
import java.util.Comparator;

class Sum_Digit implements Comparator <Integer>
    {
        @Override
        public int compare (Integer x1, Integer x2)
        {
            int t1 = 0;
            int t2 = 0;
            t1 = sum_dig(x1);
            t2 = sum_dig(x2);
               if (t1 == t2)
                return 0;
            if (t1>t2)
                return 1;
            return -1;
        }
        public int sum_dig (int x)
        {
            int t = 0;
            while (x > 0)
            {
                t = t + x%10;
                x/=10;
            }
            return t;
        }
        
    }
